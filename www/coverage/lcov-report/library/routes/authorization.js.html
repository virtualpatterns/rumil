'use strict'
 
const _Authorization = require('client-oauth2')
const Utilities = require('util')
 
const Log = require('../log')
const Path = require('../path')
const Process = require('../process')
 
class Authorization {
 
  constructor() {
  }
 
  static createRoutes(server) {
 
    server.head('/api/authorization/:system', (request, response, next) => {
      // Log.debug('- server.head(\'/api/authorization/:system\', (request, response, next) => { ... }')
      response.send(200)
      next()
    })
 
    server.get('/api/authorization/:system', (request, response, next) => {
      // Log.debug('- server.get(\'/api/authorization/:system\', (request, response, next) => { ... }\n\n%s\n', Utilities.inspect(request.params))
 
      let authorizationName = `${request.params.system}-authorization`
      let authorizationPath = Path.join(__dirname, 'systems', authorizationName)
 
      // Log.debug('> require(%j)', Path.trim(authorizationPath))
      let authorization = require(authorizationPath)
 
 
 
      if (!request.params.code) {
 
        let authorizationUri = authorization.code.getUri()
 
        Log.debug('> response.redirect(%j, next)', authorizationUri)
        response.redirect(authorizationUri, next)
 
      }
      else {
        Log.debug('> request.params.code=%j', request.params.code)
 
        authorization.code.getToken(request.url)
          .then(function (token) {
 
            Log.debug('> token.accessToken=%j', token.accessToken)
            Log.debug('> token.tokenType=%j', token.tokenType)
 
            // response.send(200)
            // next()
 
            let tokenUrl = `/www/index.html?token=${encodeURI(token.accessToken)}&type=${encodeURI(token.tokenType)}`
 
            Log.debug('> response.redirect(%j, next)', tokenUrl)
            response.redirect(tokenUrl, next)
 
          })
          .catch((error) => {
            response.send(error)
            next()
          })
 
      }
 
    })
 
  }
 
}
 
module.exports = Authorization
